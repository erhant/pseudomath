bubbleSort(A[N])
{
   do
   {
      t = 0;
      for (i=1; i < N-1; i = i+1)
      {
         if (A[i] > A[i+1])
         {
            x = A[i];
            A[i] = A[i+1];
            A[i+1] = x;
            t = 1;
         }
      }
   }
   while (t == 1);
   return A;
}
